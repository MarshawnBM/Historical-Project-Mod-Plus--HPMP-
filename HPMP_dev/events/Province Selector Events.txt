province_event = {
	id = 91707
	title = "EVTNAME91707"
	desc = "EVTDESC91707"

	trigger = {
		owner = {
			NOT = { has_country_modifier = diplomatic_cores }
		}
		has_province_modifier = removing_cores
	}

	option = {
		name = "EVTOPTA91707"
		state_scope = {
			any_owned = {
				limit = {
					has_province_modifier = removing_cores
				}
				remove_province_modifier = removing_cores
				add_core = RMC
			}
		}
		any_country = {
			limit = {
				any_core = {
					is_core = RMC
					owned_by = THIS
				}
				NOT = { tag = THIS }
			}
			country_event = { id = 91708 days = 0 }
		}
	}
}
